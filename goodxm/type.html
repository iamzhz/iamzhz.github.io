打字练习

只能运行在windows上

只有"打字练习.c"是我写的，我用了《C语言从入门到项目实战》这本书作者写的一个库，我用的32的库,也有64位的（它原来是一个.a的文件，我嫌麻烦，弄成了.o文件）。

它会先读取当前目录的1.txt文件，根据1.txt的内容练习。

Code:

  1. // 打字练习.c
  2. /*
  3.  * 编译方法:gcc 打字练习.c mycon.o -o 打字练习.exe -lwinmm
  4.  */
  5. #include <stdio.h>
  6. #include "mycon.h"
  7. #include <stdlib.h>
  8.  
  9. int main(void)
  10. {
  11. 	cls();
  12.  
  13. 	FILE *fp;
  14. 	fp=fopen("1.txt", "r");
  15.  
  16. 	set_size(30,100);                 // 宽=50，高=30 
  17. 	set_title("打字练习");
  18. 	system("color 3F");              // 2=绿色，F=白色 
  19. 	gotoxy(2,1);                     // x=2,y=3
  20.  
  21. 	char a;                          // a=接收的内容 
  22. 	char b;                          // b=文件里的一个字符 
  23. 	int c=0 ,d=0 ,f=0;               // c=错了几个 ,d=对了几个，f=一共几个 
  24.  
  25.  
  26. 	//从下一行 
  27. 	b=fgetc(fp);
  28. 	while(b != EOF)
  29. 	{
  30. 		f++ ;                        // f=一共几个 
  31. 		printf("%c", b);
  32. 		b=fgetc(fp);
  33. 	}
  34. 	//到上一行  是显示文件内容
  35.  
  36. 	gotoxy(2,2);                     // x=3,y=3
  37.  
  38. 	rewind(fp);                      // fp - 首位
  39.  
  40. 	set_color(3,5);
  41.  
  42. 	//从下一行 
  43. 	b=fgetc(fp);
  44. 	while(b != EOF)
  45. 	{
  46. 		a=getkey();
  47. 		printf("%c", a);
  48.  
  49. 		if(a != b)   c++ ;            // c=错了几个 ,d=对了几个
  50. 		else         d++ ;
  51.  
  52. 		b=fgetc(fp);
  53. 	}
  54. 	//到上一行  是接收和判断
  55.  
  56. 	set_color(3,0xF);
  57.  
  58. 	if(c == 0)
  59. 	{
  60. 		printf("\n\n\nVery good!");
  61. 	}else{
  62. 		printf("\n\n\n一共%d 个，你错了%d 个，对了%d 个。",f ,c ,d);
  63. 	}
  64. 	//无限循环空语句(可以看清结果)
  65. 	while(1){;}
  66.  
  67. 	return 0;
  68. }
  1. // mycon.h
  2. #ifndef _WINDOWS_H_
  3. #define	_WINDOWS_H_
  4. #endif
  5.  
  6. //改变控制台窗口大小为width宽，height高
  7. void set_size(int width, int height);
  8.  
  9. //设置控制台窗口的标题为字符串title
  10. void set_title(char *title);
  11.  
  12. //移动光标到坐标(x, y)处，即y行x列
  13. void gotoxy(int x, int y); 
  14.  
  15. //隐藏光标
  16. void hide_cursor(void);
  17.  
  18. //显示光标
  19. void show_cursor(void);
  20.  
  21. //设置颜色函数： bcolor背景色，fcolor前景色 
  22. //具体颜色值参见命令行下的color/?命令。
  23. void set_color(int bcolor, int fcolor);
  24.  
  25. //延时函数，单位毫秒
  26. void delay(int dtime);
  27.  
  28. //播放wav格式的声音
  29. void play_sound(const char *sndfile);
  30.  
  31. //清屏
  32. void cls(void);
  33.  
  34. //获取按键
  35. int getkey(void);
